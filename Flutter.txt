Screen Name (Single Screen Only) 
Lecture3SingleScreen 
UI Requirements 
Your screen must contain these sections (top to bottom): 
1. Header Section (Container + Text + Icon) 
ÔÇ∑ A Container with padding and rounded corners 
ÔÇ∑ A title Text: "Lecture 3: Widgets in Flutter" 
ÔÇ∑ An Icon next to the title 
2. Image Section 
ÔÇ∑ Display one image 
o Use Image.network( ) (recommended) 
3. StatelessWidget Section 
ÔÇ∑ Create a separate StatelessWidget named: InfoCard 
ÔÇ∑ It should display: 
o name 
o description 
o icon 
ÔÇ∑ Introduce yourself in the InfoCard 
4. StatefulWidget Section (Counter) 
ÔÇ∑ On the same screen, add: 
o Text: Counter: X 
o A + button to increment 
o A Reset button to set counter back to 0 
ÔÇ∑ Must update UI using setState()









import 'package:flutter/material.dart';

void main() {
  runApp(const Lecture3SingleScreen());
}

class Lecture3SingleScreen extends StatelessWidget {
  const Lecture3SingleScreen({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      debugShowCheckedModeBanner: false,
      home: HomeScreen(),
    );
  }
}

class HomeScreen extends StatefulWidget {
  const HomeScreen({super.key});

  @override
  State<HomeScreen> createState() => _HomeScreenState();
}

class _HomeScreenState extends State<HomeScreen> {
  int counter = 0;

  void incrementCounter() {
    setState(() {
      counter++;
    });
  }

  void resetCounter() {
    setState(() {
      counter = 0;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("Lecture3SingleScreen"),
        centerTitle: true,
      ),
      body: SingleChildScrollView(
        child: Column(
          children: [

            /// 1Ô∏è‚É£ Header Section
            Container(
              margin: const EdgeInsets.all(16),
              padding: const EdgeInsets.all(16),
              decoration: BoxDecoration(
                color: Colors.blue.shade100,
                borderRadius: BorderRadius.circular(15),
              ),
              child: const Row(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Icon(Icons.school, size: 28),
                  SizedBox(width: 10),
                  Text(
                    "Lecture 3: Widgets in Flutter",
                    style: TextStyle(
                      fontSize: 18,
                      fontWeight: FontWeight.bold,
                    ),
                  ),
                ],
              ),
            ),

            /// 2Ô∏è‚É£ Image Section
            Container(
              margin: const EdgeInsets.all(16),
              child: Image.network(
                "https://flutter.dev/images/flutter-logo-sharing.png",
                height: 150,
              ),
            ),

            /// 3Ô∏è‚É£ StatelessWidget Section (InfoCard)
            const InfoCard(
              name: "Saqib Younas",
              description: "I am learning Flutter and exploring Widgets.",
              icon: Icons.person,
            ),

            const SizedBox(height: 20),

            /// 4Ô∏è‚É£ Stateful Counter Section
            Text(
              "Counter: $counter",
              style: const TextStyle(fontSize: 22),
            ),

            const SizedBox(height: 10),

            Row(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                ElevatedButton(
                  onPressed: incrementCounter,
                  child: const Text("+"),
                ),
                const SizedBox(width: 15),
                ElevatedButton(
                  onPressed: resetCounter,
                  child: const Text("Reset"),
                ),
              ],
            ),

            const SizedBox(height: 30),
          ],
        ),
      ),
    );
  }
}

/// 3Ô∏è‚É£ Separate StatelessWidget
class InfoCard extends StatelessWidget {
  final String name;
  final String description;
  final IconData icon;

  const InfoCard({
    super.key,
    required this.name,
    required this.description,
    required this.icon,
  });

  @override
  Widget build(BuildContext context) {
    return Container(
      margin: const EdgeInsets.all(16),
      padding: const EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: Colors.green.shade100,
        borderRadius: BorderRadius.circular(15),
      ),
      child: Row(
        children: [
          Icon(icon, size: 40),
          const SizedBox(width: 15),
          Expanded(
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text(
                  name,
                  style: const TextStyle(
                    fontSize: 18,
                    fontWeight: FontWeight.bold,
                  ),
                ),
                Text(description),
              ],
            ),
          ),
        ],
      ),
    );
  }
}
